Booking Info table:

CREATE TABLE  `rallysite`.`BookingInfo` (
`id` INT NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`bookingSummary` TEXT NOT NULL ,
`bookingInfo` TEXT NOT NULL
) ENGINE = MYISAM ;

Fees table:

CREATE TABLE  `rallysite`.`Fees` (
`id` INT NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`fee` REAL NOT NULL ,
`deadline` DATE NOT NULL ,
`bookingInfoID` INT NOT NULL
) ENGINE = MYISAM ;

Booking table:

CREATE TABLE  `rallysite`.`Booking` (
`id` INT NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`userID` INT NOT NULL
) ENGINE = MYISAM ;
